@Override
protected void paintComponent(Graphics g) {
    super.paintComponent(g);

    Graphics2D g2d = (Graphics2D) g.create();
    g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);

    int w = getWidth();
    int h = getHeight();

    // ðŸŒˆ Gradiente base (dall'alto verso il basso)
    Color startColor = new Color(Costanti.COLOR1_HEX).brighter();
    Color endColor = new Color(Costanti.COLOR2_HEX).darker();
    GradientPaint base = new GradientPaint(0, 0, startColor, 0, h, endColor);
    g2d.setPaint(base);
    g2d.fillRect(0, 0, w, h);

    // âœ¨ Effetto glossy (riflesso bianco nella parte superiore)
    GradientPaint gloss = new GradientPaint(
        0, 0, new Color(255, 255, 255, 100),     // bianco semi-trasparente
        0, h * 0.4f, new Color(255, 255, 255, 0) // svanisce dopo il 40% dellâ€™altezza
    );
    g2d.setPaint(gloss);
    g2d.fillRect(0, 0, w, (int)(h * 0.4)); // applica solo nella parte alta

    g2d.dispose();
}
